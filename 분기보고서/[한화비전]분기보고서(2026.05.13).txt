def format_억(value_raw, unit='원'):
    """Convert raw value to 'XX억' format string."""
    # Remove commas
    if isinstance(value_raw, str):
        value_raw = value_raw.replace(',', '')
    
    value = float(value_raw)
    
    if unit == '천원':
        value_in_won = value * 1000
    elif unit == '백만원':
        value_in_won = value * 1_000_000
    elif unit == '원':
        value_in_won = value
    else:
        value_in_won = value
    
    # Convert to 억 (100,000,000 won)
    value_in_억 = value_in_won / 100_000_000
    
    # Round to nearest integer
    value_in_억 = round(value_in_억)
    
    # Format with commas
    return f"{value_in_억:,}억"

def extract_value_after_label(lines, label, context=5):
    """Find a label and return the value after it (within context lines)."""
    for i, line in enumerate(lines):
        if label in line:
            # Look for a numeric value in the next few non-empty lines
            for j in range(1, context + 1):
                if i + j < len(lines):
                    next_line = lines[i + j].strip()
                    if next_line and re.match(r'^[\d,()-]+$', next_line.replace(' ', '')):
                        return next_line
    return None

def extract_table_value(lines, start_idx, label):
    """Extract the first numeric value after a label in a table format."""
    for i in range(start_idx, min(start_idx + 10, len(lines))):
        if label in lines[i]:
            for j in range(1, 8):
                if i + j < len(lines):
                    val = lines[i + j].strip()
                    if val and not val.startswith('(') and not val.startswith('[') and not val.startswith('ㆍ'):
                        # Try to match a number
                        clean = val.replace(',', '').replace('Δ', '').strip()
                        if clean and (clean.lstrip('-').replace('.', '').isdigit() or 
                                      (clean.startswith('(') and clean.endswith(')'))):
                            return val
    return None
